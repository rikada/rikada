#importing Wordle trends data

#establishing the Wordle data as a data frame
data_df <-as.data.frame(data)
# To establish google trends relative to this dataset, I used the density of searches. I filtered to hone on a specific time frame in the dataset (this year :))
specific <- data_df %>%
